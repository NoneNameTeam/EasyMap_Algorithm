修改前示例json数据
{
    "n": 3,       //节点数量
    "start": 1,   //起点节点id
    "target": 2,  //终点节点id
    "coords": [   //coord表示每个点的坐标，后端决定坐标，预估两个节点的相对距离，从1到n不能改顺序
        {"x": 1, "y": 1},     //节点1的坐标为（1，1）
        {"x": 2, "y": 2},     //节点2的坐标为（2，2）
        {"x": 3, "y": 3}      //节点3的坐标为（3，3）
    ],
    "edges": [    //edge用于邻接表的输入，表示从u节点到v节点的代价是w,代价w可以是路程，时间等量
        {"u": 1, "v": 2, "w": 1.5},   //节点1到节点2的代价是1.5
        {"u": 2, "v": 3, "w": 1.5}    //节点2到节点3的代价是1.5
    ]
}
